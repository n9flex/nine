```typescript
import { UI } from "../../lib/ui";

  description: "Description",
  requires: [],
  inputs: ["ip"],
  outputs: ["ports", "ips"],
};

// ============================================================================
// SECTION 3: Core Logic
// ============================================================================
export async function run(
  mission: MissionManifest,
  ui: UI,
  args?: string[]
): Promise<ModuleResult> {
  // 1. Get target from args or mission
  const target = args?.[0] || getTargetFromMission(mission);
  
  // 2. Validate target

    ui.error("No target specified");

  // 3. Execute reconnaissance logic
  // 4. Return result with newAssets
  return {
    success: true,
    data: { ... },
    newAssets: [
      { type: "ip", value: "x.x.x.x", parent: target },
    ],
  };
}

// ============================================================================
// SECTION 4: Helper Functions
// ============================================================================
// Private functions for the module

// ============================================================================
// SECTION 5: Default Export
// ============================================================================
export default { meta, run };
```

### 1. lib/python.ts (Python Wrapper Helper)
**Dépendances**: `lib/ui.ts`, `lib/storage.ts`  
**Exports**: `runPythonModule`, `ensurePythonScript`

**Fonction clé**:
```typescript
export async function runPythonModule(
  scriptName: string,
  args: string[],
  ui: UI,
  options?: { timeout?: number }
): Promise<{ success: boolean; output: string; error?: string }> {
  // 1. Check python3 installed
  if (!checkLib("python3")) {
    ui.info("Installing python3...");
    await installLib("python3");
  }
  
  // 2. Download from HackDB if needed
  const cwd = await FileSystem.cwd();
  const scriptPath = `${cwd.absolutePath}/downloads/${scriptName}`;
  if (!await fileExists(scriptPath, true)) {
    await HackDB.DownloadExploit(scriptName, "downloads", { absolute: true });
  }
  
  // 3. Execute with timeout
  try {
    const output = await Shell.Process.exec(
      `python3 ${scriptPath} ${args.join(" ")}`,
      { absolute: true }
    );
    return { success: true, output };
  } catch (err) {
    return { success: false, output: "", error: String(err) };
  }
}
```

### 2. modules/recon/scanner.ts
**Dépendances**: `lib/types.ts`, `lib/ui.ts`  
**Exports**: `meta`, `run`

**APIs HackHub utilisées**:

**Logique clé**:
```typescript
export async function run(mission, ui, args): Promise<ModuleResult> {
  const target = args?.[0] || mission.assets.ips.find(ip => ip.status === "discovered")?.value;
  
  if (!target || !Networking.IsIp(target)) {
    return { success: false, data: { error: "Invalid target" } };
  }
  
  ui.info(`Scanning ${target}...`);
  
  const subnet = await Networking.GetSubnet(target);
  if (!subnet) return { success: false, data: { error: "No subnet" } };
  
  const portNumbers = await subnet.GetPorts();
  const ports: PortInfo[] = [];
  
  for (const port of portNumbers) {
    const [data, open] = await Promise.all([
      subnet.GetPortData(port),
      subnet.PingPort(port)
    ]);
    
    if (data) {
      ports.push({
        port,
        state: open ? "open" : "closed",
        service: data.service,
        version: data.version,
        forwarded: data.external !== data.internal ? {
          externalPort: data.external,
          internalPort: data.internal,
          targetIp: data.target,
        } : undefined,
      });
    }
  }
  
  // Update asset in mission
  const asset = mission.assets.ips.find(ip => ip.value === target);
  if (asset) {
    asset.status = "scanned";
    asset.ports = ports;
  }
  
  ui.success(`Found ${ports.filter(p => p.state === "open").length} open ports`);
  
  return {
    success: true,
    data: { target, ports, scannedAt: new Date().toISOString() },
    newAssets: [], // Scanner updates existing asset, doesn't create new
  };
}
```

```typescript
export async function run(mission, ui, args): Promise<ModuleResult> {
  const target = args?.[0] || ...;
  
  ui.info(`Running nettree on ${target}...`);
  
  // Use python wrapper helper
  const result = await runPythonModule("net_tree.py", [target], ui);
  
  if (!result.success) {
    ui.warn("Nettree failed, using mock discovery");
    // Fallback for testing
    const mockIps = ["192.168.1.10", "192.168.1.11"];
    return {
      success: true,
      data: { source: target, discovered: mockIps },
      newAssets: mockIps.map(ip => ({ type: "ip", value: ip, parent: target })),
    };
  }
  
  // Parse discovered IPs from output
  const ipRegex = /\b(?:\d{1,3}\.){3}\d{1,3}\b/g;
  const discoveredIps = [...new Set(result.output.match(ipRegex) || [])];
  
  ui.success(`Discovered ${discoveredIps.length} network devices`);
  
  return {
    success: true,
    data: { source: target, discovered: discoveredIps, topology: result.output },
    newAssets: discoveredIps.map(ip => ({ type: "ip", value: ip, parent: target })),
  };
}
```

## APIs HackHub Utilisées

### Networking API (pour scanner)
- `Networking.IsIp(ip: string): boolean`
- `Networking.GetSubnet(ip: string): Promise<Subnet | null>`
- `subnet.GetPorts(): Promise<number[]>`
- `subnet.GetPortData(port: number): Promise<PortData | null>`
- `subnet.PingPort(port: number): Promise<boolean>`

### Shell API (pour Python wrappers)
- `Shell.Process.exec(cmd: string, options?: { cwd?: string; absolute?: boolean }): Promise<void>`
- **Important**: Ne retourne pas de string! Output va au terminal.
- Pour capturer output: rediriger vers fichier `> output.txt` puis `FileSystem.ReadFile()`

### HackDB API (pour scripts Python)
- `HackDB.DownloadExploit(name: string, path?: string, options?: { absolute?: boolean }): Promise<void>`

## Pattern Critique: Parent Relationships

Les modules de découverte (nettree, dig) doivent définir `parent` sur les nouveaux assets:

```typescript
newAssets: [
  { type: "ip", value: "192.168.1.10", parent: target },  // discovered FROM target
]
```

Cela crée l'arbre de découverte dans le manifest.
